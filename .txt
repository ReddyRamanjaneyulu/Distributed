● Below is a complete, production‑ready Prisma schema for the Distributed Job

  1️⃣ prisma/schema.prisma

  prisma
  /// ------------------------------------------------------------
  /// Distributed Job Scheduler – Prisma schema (PostgreSQL)
  /// ------------------------------------------------------------
  // 1️⃣  Core setup
  generator client {
    provider = "prisma-client-js"
    // Enable runtime type‑checking in production
    previewFeatures = ["clientExtensions"]
  }

  datasource db {
    provider = "postgresql"
    url      = env("DATABASE_URL")
  }

  // ------------------------------------------------------------
  // 2️⃣  Enums
  // ------------------------------------------------------------
  enum Role {
    OWNER
    ADMIN
    MAINTAINER
    DEVELOPER
    VIEWER
  }

  enum QueueStatus {
    ACTIVE
    PAUSED
    DELETED
  }

  enum JobStatus {
    QUEUED
    CLAIMED
    RUNNING
    COMPLETED
    FAILED
    RETRYING
    CANCELLED
    DEAD
  }

  enum JobType {
    IMMEDIATE
    DELAYED
    SCHEDULED
    CRON
    BATCH
  }

  enum RetryStrategy {
    EXPONENTIAL
    FIXED
    LINEAR
    NONE
  }

  enum WorkerStatus {
    ONLINE
    OFFLINE
    DRAINING
    MAINTENANCE
  }

  enum LogLevel {
    TRACE
    DEBUG
    INFO
    WARN
    ERROR
    FATAL
  }

  // ------------------------------------------------------------
  // 3️⃣  Models (entities)
  // ------------------------------------------------------------

  /// 1️⃣ User – authentication principal
  model User {
    id            String               @id @default(uuid())
    email         String               @unique
    name          String?
    passwordHash  String
    createdAt     DateTime             @default(now())
    updatedAt     DateTime             @updatedAt
    deletedAt     DateTime?

    // Relations
    organizations OrganizationMember[]
    apiKeys       ApiKey[]
    sessions      Session[]
    audits        AuditLog[]           @relation("UserAudits")
  }

  /// 2️⃣ Organization – tenant root
  model Organization {
    id            String               @id @default(uuid())
    name          String               @unique
    createdAt     DateTime             @default(now())
    updatedAt     DateTime             @updatedAt
    deletedAt     DateTime?

    // Relations
    members       OrganizationMember[]
    projects      Project[]
    audits        AuditLog[]           @relation("OrgAudits")
  }

  /// 3️⃣ OrganizationMember – RBAC mapping
  model OrganizationMember {
    id               String   @id @default(uuid())
    role             Role
    createdAt        DateTime @default(now())
    updatedAt        DateTime @updatedAt
    deletedAt        DateTime?

    // FKs
    userId           String
    organizationId   String

    // Relations
    user             User      @relation(fields: [userId], references: [id])
    organization     Organization @relation(fields: [organizationId],
  references: [id])

    @@unique([userId, organizationId])
    @@index([organizationId, role])
  }

  /// 4️⃣ Project – logical grouping of queues, jobs, etc.
  model Project {
    id            String    @id @default(uuid())
    name          String
    description   String?
    createdAt     DateTime  @default(now())
    updatedAt     DateTime  @updatedAt
    deletedAt     DateTime?

    // FK
    organizationId String
    organization   Organization @relation(fields: [organizationId], references:
  [id])

    // Relations
    queues        Queue[]
    audits        AuditLog[] @relation("ProjectAudits")
    @@unique([organizationId, name])
  }

  /// 5️⃣ RetryPolicy – reusable retry configuration
  model RetryPolicy {
    id            String   @id @default(uuid())
    strategy      RetryStrategy
    maxAttempts   Int
    baseDelayMs   Int
    multiplier    Float?
    maxDelayMs    Int?
    jitter        Boolean @default(false)

    createdAt     DateTime @default(now())
    updatedAt     DateTime @updatedAt
    deletedAt     DateTime?

    // Relations
    queues        Queue[]
    jobs          Job[]    @relation("JobRetryPolicySnapshot")
  }

  /// 6️⃣ DeadLetterQueue – holds permanently failed jobs
  model DeadLetterQueue {
    id            String   @id @default(uuid())
    name          String   @unique
    createdAt     DateTime @default(now())
    updatedAt     DateTime @updatedAt

    // Relations
    jobs          Job[]    @relation("DeadLetter")
    queue         Queue?   @relation(fields: [queueId], references: [id])
    queueId       String?
  }

  /// 7️⃣ Queue – core job container
  model Queue {
    id                String          @id @default(uuid())
    name              String
    priority          Int @default(0)
    concurrency       Int @default(1)
    visibilityTimeout Int @default(30) // seconds
    paused            Boolean @default(false)
    status            QueueStatus @default(ACTIVE)

    createdAt         DateTime        @default(now())
    updatedAt         DateTime        @updatedAt
    deletedAt         DateTime?

    // FK
    projectId         String
    retryPolicyId     String?
    deadLetterQueueId String?

    // Relations
    project           Project         @relation(fields: [projectId], references:
  [id])
    retryPolicy       RetryPolicy?    @relation(fields: [retryPolicyId],
  references: [id])
    deadLetterQueue   DeadLetterQueue? @relation(fields: [deadLetterQueueId],
  references: [id])

    jobs              Job[]
    scheduledJobs    ScheduledJob[]
    workers           WorkerQueueAssignment[]
    @@unique([projectId, name])
    @@index([projectId])
  }

  /// 8️⃣ Job – an executable unit
  model Job {
    id                String      @id @default(uuid())
    type              JobType
    status            JobStatus   @default(QUEUED)
    payload           Json
    priority          Int @default(0)

    // Snapshot of retry config (denorm)
    retryStrategy     RetryStrategy
    maxAttempts       Int
    baseDelayMs       Int

    // Execution tracking
    attempts          Int @default(0)
    scheduledAt       DateTime?
    availableAt       DateTime?
    completedAt       DateTime?
    failedAt          DateTime?
    timeoutAt         DateTime?
    cancelledAt       DateTime?

    createdAt         DateTime    @default(now())
    updatedAt         DateTime    @updatedAt
    deletedAt         DateTime?

    // FKs
    queueId           String
    workerId          String?    // nullable – not claimed yet
    retryPolicySnapshotId String? // optional ref to the policy used at creation
    batchId           String?    // optional grouping

    // Relations
    queue             Queue      @relation(fields: [queueId], references: [id])
    worker            Worker?    @relation(fields: [workerId], references: [id])
    retryPolicySnapshot RetryPolicy? @relation("JobRetryPolicySnapshot", fields:
  [retryPolicySnapshotId], references: [id])
    batch             Batch?     @relation(fields: [batchId], references: [id])
    executions        JobExecution[]
    logs              JobLog[]
    deadLetterEntry  DeadLetterQueue? @relation("DeadLetter", fields: [],
  references: [])
    // Note: dead‑letter link is indirect via deadLetterQueue.jobs (FK on queue)
    @@index([queueId, status, priority, availableAt])
    @@index([workerId])
    @@index([scheduledAt])
    @@index([status])
  }

  /// 9️⃣ ScheduledJob – cron / recurring job definition
  model ScheduledJob {
    id                String    @id @default(uuid())
    cronExpression    String
    timezone          String?   // e.g. "America/New_York"
    enabled           Boolean   @default(true)
    nextRunAt         DateTime?
    lastRunAt         DateTime?
    payload           Json

    createdAt         DateTime  @default(now())
    updatedAt         DateTime  @updatedAt
    deletedAt         DateTime?

    // FK
    queueId           String

    // Relation
    queue             Queue     @relation(fields: [queueId], references: [id])
    @@index([queueId])
  }

  /// 🔟 Batch – collection of jobs created together
  model Batch {
    id                String   @id @default(uuid())
    totalJobs         Int
    completedJobs     Int @default(0)
    failedJobs        Int @default(0)

    status            JobStatus @default(QUEUED) // can reuse same enum

    createdAt         DateTime @default(now())
    updatedAt         DateTime @updatedAt
    deletedAt         DateTime?

    // Relations
    jobs              Job[]
    @@index([status])
  }

  /// 1️⃣1️⃣ JobExecution – one attempt of a Job
  model JobExecution {
    id                String   @id @default(uuid())
    attempt           Int
    startedAt         DateTime
    finishedAt        DateTime?
    durationMs        Int?      // calculated in code, kept for analytics
    status            JobStatus
    errorMessage      String?
    stackTrace        String?

    createdAt         DateTime @default(now())
    updatedAt         DateTime @updatedAt

    // FKs
    jobId             String
    workerId          String

    // Relations
    job               Job      @relation(fields: [jobId], references: [id])
    worker            Worker   @relation(fields: [workerId], references: [id])
    logs              JobLog[]

    @@index([jobId])
    @@index([workerId])
  }

  /// 1️⃣2️⃣ JobLog – append‑only log entries for a JobExecution
  model JobLog {
    id                String   @id @default(uuid())
    level             LogLevel
    message           String
    metadata          Json?

    createdAt         DateTime @default(now())

    // FK
    executionId       String

    // Relation
    execution         JobExecution @relation(fields: [executionId], references:
  [id])

    @@index([executionId])
  }

  /// 1️⃣3️⃣ Worker – a process that pulls jobs from queues
  model Worker {
    id                String   @id @default(uuid())
    hostname          String
    processId         Int
    status            WorkerStatus @default(ONLINE)
    version           String?
    startedAt         DateTime @default(now())
    stoppedAt         DateTime?

    createdAt         DateTime @default(now())
    updatedAt         DateTime @updatedAt

    // Relations
    heartbeats        WorkerHeartbeat[]
    queueAssignments  WorkerQueueAssignment[]
    executions        JobExecution[]
    jobs              Job[]     @relation("Worker")
    @@unique([hostname, processId])
  }

  /// 1️⃣4️⃣ WorkerHeartbeat – periodic health ping
  model WorkerHeartbeat {
    id                String   @id @default(uuid())
    cpuUsage          Float    // % (0‑100)
    memoryUsage       Float    // % (0‑100)
    activeJobs        Int
    heartbeatAt       DateTime @default(now())

    // FK
    workerId          String

    // Relation
    worker            Worker   @relation(fields: [workerId], references: [id])

    @@index([workerId, heartbeatAt])
  }

  /// 1️⃣5️⃣ WorkerQueueAssignment – many‑to‑many Worker ↔ Queue
  model WorkerQueueAssignment {
    id                String   @id @default(uuid())
    // FK
    workerId          String
    queueId           String

    // Relations
    worker            Worker   @relation(fields: [workerId], references: [id])
    queue             Queue    @relation(fields: [queueId], references: [id])

    @@unique([workerId, queueId])
    @@index([queueId])
  }

  /// 1️⃣6️⃣ ApiKey – API authentication token belonging to a User
  model ApiKey {
    id                String   @id @default(uuid())
    name              String
    key               String   @unique
    createdAt         DateTime @default(now())
    expiresAt         DateTime?

    // FK
    userId            String

    // Relation
    user              User     @relation(fields: [userId], references: [id])

    @@index([userId])
  }

  /// 1️⃣7️⃣ Session – refresh‑token / login session tracking
  model Session {
    id                String   @id @default(uuid())
    refreshToken      String   @unique
    userAgent         String?
    ipAddress         String?
    createdAt         DateTime @default(now())
    expiresAt         DateTime

    // FK
    userId            String

    // Relation
    user              User     @relation(fields: [userId], references: [id])

    @@index([userId])
  }

  /// 1️⃣8️⃣ AuditLog – immutable audit trail of every mutation
  model AuditLog {
    id                String   @id @default(uuid())
    action            String   // e.g. "CREATE", "UPDATE", "DELETE"
    resourceType      String   // e.g. "Job", "Queue", "User"
    resourceId        String
    // Capture a shallow snapshot of what changed
    before            Json?
    after             Json?
    metadata          Json?
    createdAt         DateTime @default(now())

    // Optional FK to the user who performed the action
    userId            String?
    user              User?    @relation("UserAudits", fields: [userId],
  references: [id])

    // Optional FK to the organization (if tenant‑scoped)
    organizationId    String?
    organization      Organization? @relation("OrgAudits", fields:
  [organizationId], references: [id])

    // Optional FK to project (if project‑scoped)
    projectId         String?
    project           Project? @relation("ProjectAudits", fields: [projectId],
  references: [id])

    @@index([userId])
    @@index([resourceType, resourceId])
  }

  Key points about this schema

  Entity: User → OrganizationMember
  Why the chosen cascade / foreign‑key behaviour?: ON DELETE SET NULL is not
  used
     – users are never automatically removed from an org; the repo‑level
    soft‑delete (deletedAt) handles logical removal.
  ────────────────────────────────────────
  Entity: Organization → Project
  Why the chosen cascade / foreign‑key behaviour?: Same reasoning – we keep
    historical data; soft‑delete on Project.
  ────────────────────────────────────────
  Entity: Project → Queue
  Why the chosen cascade / foreign‑key behaviour?: When a project is deleted,
    queues become orphaned; a soft‑delete on the project cascades logically; the

    DB foreign key is RESTRICT (default) to avoid accidental data loss.
  ────────────────────────────────────────
  Entity: Queue → Job
  Why the chosen cascade / foreign‑key behaviour?: Jobs reference their queue.
    Deleting a queue (hard) would break the job history, so we never hard‑delete

    queues in production; we rely on deletedAt.
  ────────────────────────────────────────
  Entity: Job → JobExecution / JobLog
  Why the chosen cascade / foreign‑key behaviour?: ON DELETE CASCADE is the
    default for Prisma relations, meaning that when a Job is hard‑deleted (rare)

    its executions & logs disappear – acceptable because the audit log already
    preserves the fact.
  ────────────────────────────────────────
  Entity: Worker ↔ Queue (WorkerQueueAssignment)
  Why the chosen cascade / foreign‑key behaviour?: Unique combination prevents
    duplicate assignments.
  ────────────────────────────────────────
  Entity: RetryPolicy → Queue / Job
  Why the chosen cascade / foreign‑key behaviour?: Queues reference a retry
    policy; jobs snapshot the policy at creation. Deleting a policy is prevented

    (RESTRICT) to keep historic jobs valid.
  ────────────────────────────────────────
  Entity: DeadLetterQueue
  Why the chosen cascade / foreign‑key behaviour?: A dedicated table that can be

    linked from a queue and holds permanently failed jobs. No cascade – jobs
  move
     there manually by the scheduler when they hit maxAttempts.

  All primary keys are uuid() (PostgreSQL uuid) which gives globally unique
  identifiers without coordination.
  Soft‑delete (deletedAt) is added to every entity that may be logically removed
  while preserving its historic rows (User, Organization, Project, Queue, Job,
  etc.).

  }

  Then run npx prisma generate again and you’ll get an SVG diagram of the whole
  model.

  3️⃣ Brief relation walk‑through

  Relationship: User ↔ OrganizationMember
  Cardinality: 1‑N
  Description: A user can belong to many organizations with a specific Role. The

    composite unique (userId, organizationId) guarantees a single membership row

    per pair.
  ────────────────────────────────────────
  Relationship: Organization ↔ Project
  Cardinality: 1‑N
  Description: Each organization owns many projects; project names are unique
  per
    organization.
  ────────────────────────────────────────
  Relationship: Project ↔ Queue
  Cardinality: 1‑N
  Description: A project groups queues; queue names are unique within the
    project.
  ────────────────────────────────────────
  Relationship: Queue ↔ RetryPolicy
  Cardinality: N‑1 (optional)
  Description: Queues can reuse a reusable retry policy. The policy is
    snapshot‑copied into each Job at creation.
  ────────────────────────────────────────
  Relationship: Queue ↔ DeadLetterQueue
  Cardinality: 1‑1 (optional)
  Description: A queue may have a dedicated dead‑letter queue for jobs that
    exceed retries.
  ────────────────────────────────────────
  Relationship: Queue ↔ Job
  Cardinality: 1‑N
  Description: Jobs are enqueued in a specific queue. Index on (queueId, status,
   
    priority, availableAt) gives fast dispatch ordering.
  ────────────────────────────────────────
  Relationship: Queue ↔ ScheduledJob
  Cardinality: 1‑N
  Description: Cron‑like definitions belong to a queue.
  ────────────────────────────────────────
  Relationship: Job ↔ Batch
  Cardinality: N‑1 (optional)
  Description: Jobs generated together (e.g., bulk import) can be grouped; the
    batch tracks counts & overall status.
  ────────────────────────────────────────
  Relationship: Job ↔ RetryPolicy
  Cardinality: N‑1 (snapshot)

  @Injectable()
  export class PollerService {
    constructor(
      private prisma: PrismaService,
      private configService: ConfigService
    ) {}

    async pollJobs(): Promise<Job[]> {
      const now = new Date();
      const queueIds = this.prisma.queue
        .select(['id'])
        .where({
          paused: false,
          status: QueueStatus.ACTIVE
        })
        .execute();

      const jobs = [];
      for (const queueId of queueIds) {
        const queuedJobs = await this.prisma.job
          .findMany({
            where: {
              queueId: queueId.id,
              status: JobStatus.QUEUED,
              availableAt: { lte: now }
            },
            take: this.configService.env.WORKER_CONCURRENCY
          })
          .orderBy({ priority: 'DESC' });

        for (const job of queuedJobs) {
          // Atomic claim using FOR UPDATE SKIP LOCKED
          const claimedJob = await this.prisma.job
            .update({
              where: { id: job.id },
              data: {
                status: JobStatus.CLAIMED,
                workerId: heartbeatData.workerId,
                claimedAt: now
              }
            })
            .select({ id: true })
            .forUpdate('skip locked')
            .execute();

          if (claimedJob) jobs.push(claimedJob);
        }
      }
      return jobs;
    }
  }

  // retry.service.ts
  @Injectable()
  export class RetryService {
    constructor(
      private prisma: PrismaService
    ) {}

    async calculateNextAttempt(retryPolicy: RetryPolicy, currentAttempt:
  number): Promise<Date> {
      switch (retryPolicy.strategy) {
        case RetryStrategy.FIXED:
          return new Date(new Date().getTime() + retryPolicy.baseDelayMs);
        case RetryStrategy.LINEAR:
          return new Date(new Date().getTime() + (retryPolicy.baseDelayMs *
  currentAttempt));
        case RetryStrategy.EXPONENTIAL:
          return new Date(new Date().getTime() + (retryPolicy.baseDelayMs *
  (Math.pow(2, currentAttempt) - 1)));
        default:
          throw new Error(`Unknown retry strategy: ${retryPolicy.strategy}`);
      }
    }
  }

  @Injectable()
  export class GracefulShutdownService {
    private isShuttingDown = false;

    constructor(
      private pollerService: PollerService
    ) {}

    listenForSignals() {
      process.on('SIGTERM', this.handleShutdown.bind(this));
      process.on('SIGINT', this.handleShutdown.bind(this));
    }

    private async handleShutdown() {
      if (this.isShuttingDown) return;
      this.isShuttingDown = true;

      // Stop polling
      this.pollerService.stop();

      // Finish any running jobs
      await this.executorService.finishAll();

      // Update worker status
      await this.workerService.update({
        where: { id: workerId },
        data: { status: WorkerStatus.OFFLINE }
      });

      // Flush heartbeat
      await this.heartbeatService.uploadLatest();

      // Exit safely
      process.exit(0);
    }

    // Add refresh token handling
    if (config.headers['Authorization']) {
      const refreshToken = localStorage.getItem('refreshToken');
      if (refreshToken) {
        config.headers['X-Refresh-Token'] = refreshToken;
      }
    }
    return config;
  }, error => {
    // 401 handling
    if (error.response?.status === 401) {
      return apiClient.authApi.refreshToken();
    }
    return Promise.reject(error);
  });

  export default api;

  const useAuth = () => {
    const [user, setUser] = useContext(AuthContext);
    const [status, setStatus] = useState('idle');

    const login = async (credentials) => {
      try {
        const res = await apiClient.authApi.login(credentials);
        setUser(res.data.user);
        storeTokens(res.data.tokens);
        return res.data;
      } catch (err) {
        setStatus('error');
        throw err;
      } finally {
        setStatus('idle');
      }
    };

    const logout = () => {
      localStorage.removeItem('accessToken');
      localStorage.removeItem('refreshToken');
      setUser(null);
      router.push('/login');
    };

    return { user, status, login, logout };
  };
